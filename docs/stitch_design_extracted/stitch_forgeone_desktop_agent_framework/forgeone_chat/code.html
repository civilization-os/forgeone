import json
import urllib.request
from typing import List, Dict

def fetch_mcp_metrics(endpoint_url: str) -> List[Dict]:
    """Fetches metric data from the MCP cluster."""
    try:
        req = urllib.request.Request(endpoint_url, headers={'Accept': 'application/json'})
        with urllib.request.urlopen(req) as response:
            data = json.loads(response.read().decode())
            return data.get('metrics', [])
    except Exception as e:
        print(f"Error fetching metrics: {e}")
        return []

def format_markdown_table(metrics: List[Dict]) -> str:
    if not metrics: return "No data available."
    
    headers = list(metrics[0].keys())
    header_row = "| " + " | ".join(headers) + " |"
    separator_row = "|" + "|".join(["---" for _ in headers]) + "|"
    
    rows = []
    for m in metrics:
        rows.append("| " + " | ".join(str(m.get(h, '')) for h in headers) + " |")
        
    return "\n".join([header_row, separator_row] + rows)
